Instalação Express

# Copia este YAML para o Custom Install da tua App Store

name: syncpulse-beta
services:
  syncpulse:
    container_name: SyncPulse_Final
    image: ghcr.io/syncpulsegeral-collab/syncpulse:beta-1.0
    privileged: true
    user: root
    restart: always
    environment:
      - PYTHONUNBUFFERED=1
      - TZ=Europe/Lisbon
    ports:
      - "8181:8181"
    volumes:
      - /DATA/AppData/SyncPulse/config:/config
      - /DATA/AppData/SyncPulse/app:/app
      - /DATA/AppData/SyncPulse/www:/www
      - /mnt:/mnt
      - /etc/machine-id:/etc/machine-id:ro
